\problem{1}
\begin{wts}
    Cellular net, $\alpha=3.8$, minimum freq. reuse fctr $N\geq 7$, find required $SIR$.
\end{wts}
\begin{proof}
    $SIR = \frac{(3N)^{\alpha/2}}{L}$ so $SIR\geq 54.208$, set $L=6$ for 1st order approximation.
\end{proof}
